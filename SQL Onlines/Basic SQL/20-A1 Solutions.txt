Solution 1: 
SELECT L1.CITY, COUNT(*) "Employee Count"
FROM HR.EMPLOYEES E1 
    JOIN HR.DEPARTMENTS D1 ON E1.DEPARTMENT_ID = D1.DEPARTMENT_ID 
    JOIN HR.LOCATIONS L1 ON D1.LOCATION_ID = L1.LOCATION_ID
GROUP BY L1.CITY
ORDER BY L1.CITY 

Solution 2: 
SELECT E1.EMPLOYEE_ID, (E1.FIRST_NAME || ' ' || E1.LAST_NAME) NAME, (E2.FIRST_NAME || ' ' || E2.LAST_NAME) "Manager Name", D1.DEPARTMENT_NAME, J1.JOB_TITLE 
FROM HR.EMPLOYEES E1 
    LEFT OUTER JOIN HR.EMPLOYEES E2 ON E1.MANAGER_ID = E2.EMPLOYEE_ID 
    LEFT OUTER JOIN HR.DEPARTMENTS D1 ON E1.DEPARTMENT_ID = D1.DEPARTMENT_ID 
    LEFT OUTER JOIN HR.JOBS J1 ON E1.JOB_ID = J1.JOB_ID 
ORDER BY E1.EMPLOYEE_ID 


Solution 3: 
SELECT E2.EMPLOYEE_ID "Manager ID", (E2.FIRST_NAME || ' ' || E2.LAST_NAME) "Manager Name", E2.SALARY, Count(*) as EMPLOYEE_COUNT, ROUND(AVG(E1.SALARY),2) AVG_SALARY 
FROM HR.EMPLOYEES E1 
	JOIN HR.EMPLOYEES E2 ON (E1.MANAGER_ID = E2.EMPLOYEE_ID)
GROUP BY E2.EMPLOYEE_ID, E2.SALARY, E2.FIRST_NAME, E2.LAST_NAME
ORDER BY E2.EMPLOYEE_ID


Solution 4: 
SELECT E1.EMPLOYEE_ID, (E1.FIRST_NAME || ' ' || E1.LAST_NAME) FULLNAME, D1.DEPARTMENT_NAME, J1.JOB_TITLE, (E1.SALARY * 12 * (1 + NVL(COMMISSION_PCT, 0))) "ANNUAL SALARY"
FROM HR.EMPLOYEES E1 
    LEFT OUTER JOIN HR.JOB_HISTORY JH1 ON (JH1.EMPLOYEE_ID = E1.EMPLOYEE_ID) 
    LEFT OUTER JOIN HR.DEPARTMENTS D1 ON E1.DEPARTMENT_ID = D1.DEPARTMENT_ID 
    LEFT OUTER JOIN HR.JOBS J1 ON E1.JOB_ID = J1.JOB_ID 
WHERE (JH1.JOB_ID IS NULL) AND (JH1.DEPARTMENT_ID IS NULL)
ORDER BY E1.EMPLOYEE_ID 

Solution 5: 
SELECT E1.MANAGER_ID, (E2.FIRST_NAME || ' ' || E2.LAST_NAME) NAME, E2.SALARY, COUNT(DISTINCT E3.EMPLOYEE_ID) "RICHER THAN", COUNT(DISTINCT E4.EMPLOYEE_ID) "POORER THAN" 
FROM HR.EMPLOYEES E1 
    JOIN HR.EMPLOYEES E2 ON (E1.MANAGER_ID = E2.EMPLOYEE_ID) 
    LEFT OUTER JOIN HR.EMPLOYEES E3 ON (E2.SALARY > E3.SALARY)
    LEFT OUTER JOIN HR.EMPLOYEES E4 ON (E2.SALARY < E4.SALARY)
GROUP BY E1.MANAGER_ID, E2.SALARY, E2.FIRST_NAME, E2.LAST_NAME 
ORDER BY E1.MANAGER_ID
